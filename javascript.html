⚡ JavaScript - Desarrollo Web

JavaScript es el lenguaje de programación que corre en todos los navegadores. Es esencial para crear sitios web dinámicos e interactivos. Además de frontend, JavaScript también se usa en backend con Node.js, convirtiéndose en una herramienta fullstack poderosa.

🎯 ¿Por Qué JavaScript?

  • Ubicuo: Corre en cualquier navegador
  • Fullstack: Frontend y backend con Node.js
  • Frameworks potentes: React, Vue, Angular
  • Verificación del lado cliente: Valida datos antes de enviar
  • Experiencia de usuario: Interactividad sin recargar la página

📝 Sintaxis Básica

Variables

// var (antigua, evitar)
var nombre = "Juan";

// let (variable mutable)
let edad = 28;

// const (constante, inmutable)
const PI = 3.14159;

// Tipos de datos
typeof nombre     // string
typeof edad       // number
typeof true       // boolean
typeof {}         // object
typeof null       // object (bug histórico)
typeof undefined  // undefined

Operadores

// Aritméticos
10 + 5        // 15
10 - 5        // 5
10 * 5        // 50
10 / 5        // 2
10 % 3        // 1

// Comparación
10 > 5        // true
10 == "10"    // true (comparar valor)
10 === "10"   // false (comparar tipo)

// Lógicos
true && false       // false
true || false       // true
!true              // false

Condicionales

if (edad >= 18) {
    console.log("Mayor de edad");
} else if (edad >= 13) {
    console.log("Adolescente");
} else {
    console.log("Niño");
}

// Operador ternario
status = edad >= 18 ? "Mayor" : "Menor";

🔄 Bucles

// For tradicional
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// While
let contador = 0;
while (contador < 5) {
    console.log(contador);
    contador++;
}

// For...of (valores)
for (let valor of [1, 2, 3]) {
    console.log(valor);
}

// ForEach
[1, 2, 3].forEach(num => {
    console.log(num);
});

🔧 Funciones

// Función tradicional
function sumar(a, b) {
    return a + b;
}

// Arrow function (moderna)
const sumar = (a, b) => a + b;

// Función con parámetro por defecto
const saludar = (nombre = "Usuario") => {
    console.log(`Hola, ${nombre}`);
}

// Función que retorna función
const multiplicar = (x) => (y) => x * y;
mult3 = multiplicar(3);
console.log(mult3(5));  // 15

📊 Objetos y Arrays

Arrays

const numeros = [1, 2, 3, 4, 5];

// Acceso
console.log(numeros[0]);      // 1
console.log(numeros.length);  // 5

// Métodos
numeros.push(6);              // Añadir al final
numeros.pop();                // Quitar del final
numeros.map(n => n * 2);      // Transformar
numeros.filter(n => n > 2);   // Filtrar
numeros.find(n => n === 3);   // Buscar

Objetos

const persona = {
    nombre: "Juan",
    edad: 28,
    ciudad: "Madrid",
    saludar: function() {
        console.log(`Hola, me llamo ${this.nombre}`);
    }
};

// Acceso
persona.nombre        // "Juan"
persona["edad"]       // 28
persona.saludar();    // Hola, me llamo Juan

// Propiedades dinámicas
persona.profesion = "Developer";
delete persona.ciudad;

🔌 Fetch API (Peticiones HTTP)

// GET
fetch("https://api.ejemplo.com/datos")
    .then(respuesta => respuesta.json())
    .then(datos => console.log(datos))
    .catch(error => console.log(error));

// POST
fetch("https://api.ejemplo.com/datos", {
    method: "POST",
    headers: {
        "Content-Type": "application/json"
    },
    body: JSON.stringify({
        nombre: "Juan",
        edad: 28
    })
})
.then(respuesta => respuesta.json())
.then(datos => console.log(datos));

// Async/await (moderno)
async function obtenerDatos() {
    try {
        const respuesta = await fetch("https://api.ejemplo.com/datos");
        const datos = await respuesta.json();
        console.log(datos);
    } catch (error) {
        console.log(error);
    }
}
