1. Fundamentos SQL Avanzado

2. SELECT Avanzado

SELECT con DISTINCT

Obtener valores únicos sin duplicados.

-- Obtener todas las ciudades diferentes
SELECT DISTINCT ciudad FROM clientes;

-- Con múltiples columnas
SELECT DISTINCT ciudad, pais FROM clientes;

SELECT con WHERE complejo

-- Múltiples condiciones
SELECT * FROM productos 
WHERE precio > 100 
  AND categoria = 'Electrónica' 
  OR descuento > 0.2;

-- IN (pertenencia)
SELECT * FROM empleados 
WHERE departamento IN ('Ventas', 'IT', 'Marketing');

-- BETWEEN (rango)
SELECT * FROM pedidos 
WHERE fecha BETWEEN '2024-01-01' AND '2024-12-31';

-- LIKE (búsqueda de texto)
SELECT * FROM clientes 
WHERE nombre LIKE 'Juan%';  -- Comienza con Juan
SELECT * FROM clientes 
WHERE email LIKE '%@gmail.com';  -- Termina con @gmail.com

ORDER BY y LIMIT

-- Ordenar por una columna
SELECT * FROM productos 
ORDER BY precio DESC;  -- DESC = descendente, ASC = ascendente

-- Múltiples columnas
SELECT * FROM ventas 
ORDER BY fecha DESC, monto ASC;

-- LIMIT (limitar resultados)
SELECT * FROM productos 
ORDER BY ventas DESC 
LIMIT 10;  -- Top 10 productos más vendidos

-- LIMIT con OFFSET
SELECT * FROM usuarios 
LIMIT 10 OFFSET 20;  -- Registros 21 a 30 (paginación)

GROUP BY y HAVING

-- Agrupar por departamento
SELECT departamento, COUNT(*) as empleados 
FROM empleados 
GROUP BY departamento;

-- Agrupar con múltiples columnas
SELECT ciudad, pais, COUNT(*) as total 
FROM clientes 
GROUP BY ciudad, pais;

-- HAVING (filtro después de GROUP BY)
SELECT departamento, AVG(salario) as salario_promedio 
FROM empleados 
GROUP BY departamento 
HAVING AVG(salario) > 50000;

3. JOINS - Unir Tablas

INNER JOIN

Retorna solo registros que coinciden en ambas tablas.

SELECT p.nombre, c.nombre as cliente 
FROM pedidos p 
INNER JOIN clientes c ON p.cliente_id = c.id;

LEFT JOIN (LEFT OUTER JOIN)

Todos los registros de la izquierda + coincidencias de la derecha.

SELECT c.nombre, COUNT(p.id) as cantidad_pedidos 
FROM clientes c 
LEFT JOIN pedidos p ON c.id = p.cliente_id 
GROUP BY c.id;

RIGHT JOIN (RIGHT OUTER JOIN)

Todos los registros de la derecha + coincidencias de la izquierda.

SELECT c.nombre, p.fecha 
FROM clientes c 
RIGHT JOIN pedidos p ON c.id = p.cliente_id;

FULL OUTER JOIN

Todos los registros de ambas tablas (no soportado en MySQL).

-- MySQL: simular con UNION
SELECT c.nombre, p.fecha 
FROM clientes c 
LEFT JOIN pedidos p ON c.id = p.cliente_id 
UNION 
SELECT c.nombre, p.fecha 
FROM clientes c 
RIGHT JOIN pedidos p ON c.id = p.cliente_id;

CROSS JOIN

Producto cartesiano: cada fila de la izquierda con cada fila de la derecha.

SELECT c.nombre, p.nombre 
FROM clientes c 
CROSS JOIN productos p;  -- Combinación de todos los clientes con todos los productos

Self Join

Unir una tabla consigo misma.

-- Obtener jefe de cada empleado
SELECT e.nombre as empleado, j.nombre as jefe 
FROM empleados e 
LEFT JOIN empleados j ON e.jefe_id = j.id;

Múltiples JOINs

SELECT 
    p.id as pedido_id,
    c.nombre as cliente,
    pr.nombre as producto,
    dp.cantidad
FROM pedidos p
INNER JOIN clientes c ON p.cliente_id = c.id
INNER JOIN detalles_pedido dp ON p.id = dp.pedido_id
INNER JOIN productos pr ON dp.producto_id = pr.id;

4. Funciones de Agregación

COUNT, MIN, MAX, AVG, SUM

-- COUNT (contar registros)
SELECT COUNT(*) as total_usuarios FROM usuarios;
SELECT COUNT(DISTINCT ciudad) as ciudades FROM clientes;

-- MIN y MAX (mínimo y máximo)
SELECT MIN(precio) as precio_minimo, MAX(precio) as precio_maximo 
FROM productos;

-- AVG (promedio)
SELECT AVG(salario) as salario_promedio 
FROM empleados;

-- SUM (suma)
SELECT SUM(monto) as total_vendido 
FROM ventas 
WHERE año = 2024;

Window Functions (SQL 2012+)

Funciones que operan sobre un "ventana" de filas.

-- ROW_NUMBER: número de fila
SELECT nombre, salario,
       ROW_NUMBER() OVER (ORDER BY salario DESC) as rango
FROM empleados;

-- RANK: ranking con empates
SELECT nombre, salario,
       RANK() OVER (ORDER BY salario DESC) as rango
FROM empleados;

-- DENSE_RANK: ranking sin gaps
SELECT nombre, salario,
       DENSE_RANK() OVER (ORDER BY salario DESC) as rango
FROM empleados;

-- LAG/LEAD: acceder a filas anteriores/siguientes
SELECT 
    fecha, 
    monto,
    LAG(monto) OVER (ORDER BY fecha) as monto_anterior,
    LEAD(monto) OVER (ORDER BY fecha) as monto_siguiente
FROM ventas;

-- SUM con OVER
SELECT 
    fecha,
    monto,
    SUM(monto) OVER (ORDER BY fecha) as monto_acumulado
FROM ventas;

5. Subconsultas (Subqueries)

Subconsultas en SELECT

-- Obtener cliente y total de sus pedidos
SELECT 
    nombre,
    (SELECT COUNT(*) FROM pedidos WHERE cliente_id = clientes.id) as total_pedidos
FROM clientes;

Subconsultas en WHERE

-- Productos más caros que el promedio
SELECT * FROM productos 
WHERE precio > (SELECT AVG(precio) FROM productos);

-- Clientes que han realizado pedidos
SELECT * FROM clientes 
WHERE id IN (SELECT cliente_id FROM pedidos);

Subconsultas con ANY/ALL

-- ANY: comparación con al menos un valor
SELECT * FROM empleados 
WHERE salario > ANY (SELECT salario FROM empleados WHERE departamento = 'IT');

-- ALL: comparación con todos los valores
SELECT * FROM empleados 
WHERE salario > ALL (SELECT salario FROM empleados WHERE departamento = 'Recursos Humanos');

Subconsultas Correlacionadas

Subconsulta que referencia columnas de la consulta externa.

-- Empleados con salario mayor al promedio de su departamento
SELECT * FROM empleados e1
WHERE salario > (
    SELECT AVG(salario) 
    FROM empleados e2 
    WHERE e1.departamento = e2.departamento
);

6. Procedimientos Almacenados y Triggers

Creación de Procedimiento Almacenado

-- MySQL
DELIMITER //
CREATE PROCEDURE ObtenerClientesPorCiudad(IN p_ciudad VARCHAR(100))
BEGIN
    SELECT * FROM clientes WHERE ciudad = p_ciudad;
END //
DELIMITER ;

-- Ejecutar
CALL ObtenerClientesPorCiudad('Madrid');

Procedimiento con OUT

DELIMITER //
CREATE PROCEDURE ContarPedidos(IN p_cliente_id INT, OUT p_total INT)
BEGIN
    SELECT COUNT(*) INTO p_total FROM pedidos WHERE cliente_id = p_cliente_id;
END //
DELIMITER ;

-- Ejecutar
CALL ContarPedidos(1, @total);
SELECT @total;

Transacciones

-- Garantizar integridad: todo o nada
START TRANSACTION;

UPDATE cuentas SET saldo = saldo - 1000 WHERE id = 1;
UPDATE cuentas SET saldo = saldo + 1000 WHERE id = 2;

COMMIT;  -- Guardar cambios
-- ROLLBACK;  -- Deshacer si hay error

Triggers

Acciones automáticas cuando ocurren eventos en BD.

-- Trigger: actualizar precio_actualizado cuando se inserta producto
DELIMITER //
CREATE TRIGGER actualizar_timestamp_producto
BEFORE INSERT ON productos
FOR EACH ROW
BEGIN
    SET NEW.precio_actualizado = NOW();
END //
DELIMITER ;

Índices y Optimización

Crear Índices

-- Índice simple
CREATE INDEX idx_email ON usuarios(email);

-- Índice único (garantiza no duplicados)
CREATE UNIQUE INDEX idx_usuario_email ON usuarios(email);

-- Índice compuesto (multiple)
CREATE INDEX idx_ciudad_pais ON clientes(ciudad, pais);

-- Ver índices
SHOW INDEXES FROM usuarios;

EXPLAIN - Analizar queries

-- Ver cómo la BD ejecuta la query
EXPLAIN SELECT * FROM productos WHERE categoria = 'Electrónica';

Buenas Prácticas SQL

  • Use alias significativos: SELECT nombre AS nombre_cliente, email AS correo
  • Evite SELECT *: especifique columnas necesarias
  • Use índices en columnas de búsqueda frecuente
  • Normalice la BD para evitar redundancia
  • Use prepared statements para evitar SQL injection
  • Documente queries complejas con comentarios
  • Use transacciones para operaciones críticas
  • Realice backups regularmente
